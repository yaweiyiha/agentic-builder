/* SPRI Monitor Design Tokens — from index_risk.html */ :root { /* ─ Backgrounds ─ */ --bg: #F0F2F5; /* page background */ --surface: #FFFFFF; /* cards / panels */ --border: #E4E7EE; /* borders / dividers */ /* ─ Brand / Primary ─ */ --teal: #0BBFB2; /* primary CTA, active states */ --teal-light: #E6FAF8; /* active nav bg, badge bg */ --teal-dark: #089C91; /* hover state, mono text */ /* ─ Risk Semantic ─ */ --risk-normal: #10B981; --risk-normal-light: #D1FAE5; --risk-normal-dark: #065F46; --risk-elevated: #F59E0B; --risk-elevated-light: #FEF3C7; --risk-elevated-dark: #92400E; --risk-high: #F97316; --risk-high-light: #FFEDD5; --risk-high-dark: #7C2D12; --risk-critical: #EF4444; --risk-critical-light: #FEE2E2; --risk-critical-dark: #991B1B; --risk-mover: #8B5CF6; --risk-mover-light: #EDE9FE; --risk-mover-dark: #5B21B6; /* ─ Text ─ */ --text-primary: #111827; --text-secondary: #6B7280; --text-muted: #9CA3AF; /* ─ Shadows ─ */ --shadow-sm: 0 1px 3px rgba(0,0,0,.08), 0 1px 2px rgba(0,0,0,.05); --shadow-md: 0 4px 12px rgba(0,0,0,.08); /* ─ Radius ─ */ --radius-sm: 8px; /* buttons, inputs, small cards */ --radius-md: 12px; /* standard cards */ --radius-lg: 16px; /* modal, large hero */ }
